\begin{algorithm}[!ht]
   \small
        \caption{Algorithm $\mathcal{A}(Y, \{ h_1, \dots, h_{n_r}\}; \beta)$}
     \label{sim:frost-interactive-parallel}
         \hspace*{\algorithmicindent} \textbf{Input} A public key $Y$ and
         random oracle outputs   $\{ h_1, \dots, h_{n_r}\}$\\
         \hspace*{\algorithmicindent} \textbf{Output} An index $J$ and forgery $\sigma$, or $\bot$ \\
        \begin{algorithmic}[1]
          \STATE Initialize $ctr = 1, T_{\rho} = \{ \}, T_2 = \{ \}, T_3 = \{ \}, T_4 = \{ \}, J_2 = \{\}, C = \{\}, M = \{\}$
          \STATE Run $\mathcal{F}$ on input $Y$, answering its queries as
          follows, until it outputs $(m, \sigma=(R, z))$ or $\bot$.
          %
          % Simulation of random oracles
          \STATE \textbf{On simulating $H_{\rho}(m, B)$:}
          \STATE \hskip0.5em If $T_{\rho}[m, B] = \bot$,  set $T_{\rho}[m, B] =
          h_{ctr};
          ctr = ctr + 1$. Return $T_{\rho}[m, B]$.
          \STATE \textbf{On simulating $H_2(R, Y, m)$:}
          \STATE \hskip0.5em If $T_2[m, R] = \bot$,  set $T_2[R, Y, m] = h_{ctr}$,
          $J_2[R, Y, m] = ctr;
          ctr = ctr + 1$. Return $T_2[R, Y, m]$.
          \STATE \textbf{On simulating $H_3(\vec{X})$:}
          \STATE \hskip0.5em If $T_3[\vec{X}] = \bot$, set $T_3[\vec{X}] = h_{ctr};
          ctr = ctr + 1$. Return $T_3[\vec{X}]$.
          \STATE \textbf{On simulating $H_4(i, \Phi)$:}
          \STATE \hskip0.5em If $T_4[i, \Phi] = \bot$, set $T_4[i, \Phi] =
          h_{ctr}; ctr = ctr + 1$. Return $T_4[i, \Phi]$.
          %
          % Simulation of Pre-process
          \STATE \textbf{On simulating Preprocess:}
          %
          \STATE \hskip0.5em \textbf{Round 1:}
          \STATE \hskip1em For $1 \le j \le \pi$, do:
          \STATE \hskip1.5em Set $\bar{c}_j = h_{ctr}$, $C[j] = ctr$, $ctr = ctr + 1$,
          $\bar{z}_{tj} \random \mathbb{Z}_q$,
          $D_{tj} = g^{\bar{z}_{tj}} \cdot {Y_t}^{-\bar{c}_j}$.
          \STATE \hskip1.5em Follow the protocol honestly to sample
          $(e_{tj}, a_{tj}, b_{tj})$ and
          derive $(E_{tj}, A_{tj}, B_{tj})$.
          \STATE \hskip1.5em Follow the protocol honestly to sample
          $(k_{atj}, k_{btj})$ and derive $(R_{atj}, R_{btj})$.
          \STATE \hskip1em Derive $K_t$ honestly, publish to $\mathcal{F}$, and
          wait for all $K_\ell$ values from $\mathcal{F}$.
          %
          \STATE \hskip0.5em \textbf{Round 2:}
          \STATE \hskip1em Derive $L_t, \Phi, J_t$ honestly.
          Send $(t, L_t, J_t)$ to $\mathcal{F}$, and wait to receive the
          $(\ell, L_\ell, J_\ell)$ tuples from $\mathcal{F}$, following the
          protocol for validation.
          \STATE \hskip1em Reprogram $T_3[K_1, \dots, K_n] = h_{ctr}$; set $ctr
          = ctr + 1$. Rederive $c_t$ and $J_t$ honestly.
          \STATE \hskip1em Rewind $\mathcal{F}$ to step $1$ in Round $2$ of
          Figure~\ref{fg:interactive-protocol-preprocess},
            immediately before $\mathcal{F}$ queries $H_3$ with $(K_1, \dots,
            K_n)$.
            % CK I don't think it matters enough to the protocol to distinguish
            % H_3 more.
          \STATE \hskip1em After allowing $\mathcal{F}$ to proceed after
          rewinding, use its two sets of outputs to derive the discrete logarithm
          of each $A_{\ell j}$ and $B_{\ell j}$; store for use in the signing protocol.
          \STATE \hskip1em Complete the protocol honestly.
          %
          % Simulation of Signing
          \STATE \textbf{On simulating Sign:}
          %
          \STATE \hskip0.5em \textbf{Round 1:} Input $(m, B)$
          \STATE \hskip1em Insert $m$ into $M$.
          \STATE \hskip1em Using $(a_{\ell j},
          b_{\ell j})$ obtained during Preprocess, derive $\rho_\ell  : \ell  \in S, \ell
          \neq t$
          \STATE \hskip1em Derive $\rho_t = a_{tj} + b_{tj} \cdot H_{\rho}(m,
          B)$ and $R$, following the protocol honestly for validation.
          \STATE \hskip1em  Program $T_2[m, R] = \bar{c}_j$, $J_2[m,R] = C[j]$; return $\rho_t$.
          \STATE \hskip0.5em \textbf{Round 2:} Input $(\rho_j, \dots, \rho_t)$
          \STATE \hskip1em  Let $z_t = \bar{z}_{tj} + (e_{tj} \cdot \rho_t)$;
          return $z_t$ to $\mathcal{F}$
          %
          \STATE If $\mathcal{F}$ outputs $\bot$, then return $\bot$.  Else
          $\mathcal{F}$ outputs $(m, \sigma = (R,z))$.
          \STATE If $T_2[m,Y,R] = \bot$,  set $T_2[m,Y,R] = h_{ctr}$,
          $J_2[m,Y,R] = ctr$,
          and $ctr = ctr + 1$.
          \STATE Let $c = T_2[m,Y,R]$.  If $R \not= g^z Y^{-c}$
          or $m \in M$, then return $\bot$
          \STATE Let $J = J_2[m,Y,R]$.
          Return $J, \sigma = (R,z)$
   \end{algorithmic}
\end{algorithm}
