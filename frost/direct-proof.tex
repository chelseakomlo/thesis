\section{Security of FROST}
\label{frost:direct-proof}

We now present a direct proof for the version of FROST given in Figure~\ref{fig:frost}.
This proof builds on prior work for prior proofs of security for FROST~\cite{BellareCKMTZ22},
but gives a reduction to AOMDL, as opposed to OMDL.

\begin{theorem}\label{thm:FROST-aomdl}
	FROST is statically unforgeable against an adversary $\adv$ that is allowed to corrupt up to $\tcor$ participants,
  assuming the AOMDL assumption holds for $\GroupGen$ in the ROM.
  In other words, for every static unforgeability adversary $\adv$ that wins the adaptive unforgeability game against FROST,
  there exists a PPT adversary $\B$ that wins the AOMDL game.

  The advantage of $\adv$ is bounded by


  \begin{equation*} \label{eq:frost-aomdl}
\advant{$\uf$}{\A,\frost}(\secp, n, \thresh, \tcor)
    \leq \sqrt{\advant{aomdl}{\B}(\secp) + \mathsf{negl}(\secp) }
  \end{equation*}

such that $\numsignq$ is the number of times $\adv$ is allowed to query the signing oracles.

\end{theorem}

To prove Theorem~\ref{thm:FROST-aomdl},
we rely on two lemmas.

\begin{lemma} \label{lem:frost-simulation}
There exists an algorithm $\SIM$ which can simulate FROST with respect to an AOMDL instance to an adversary $\adv$
playing against the unforgeability game as in Figure~\ref{fig:euf-cma-threshold}.
In other words,
$\adv$ has negligible additional advantage in winning the unforgeability game when
interacting with $\SIM$ that it does when playing against FROST directly.

Concretely, in the setting where $\adv$ is allowed up to $\tcor$ corruptions, then
\begin{align} \label{eq:simulation-frost}
  \begin{split}
\advant{$\uf$}{\A,\frost}(\secp, n, \thresh, \tcor)  & \leq
        \advant{uf}{\SIM, \adv}(\lambda, n, t, t-1)
        \frac{(\honetotal + \qchal)}{q}
        + \frac{\rototal^2}{2q}
  \end{split}
\end{align}
\end{lemma}

\begin{lemma} \label{lem:frost-reduction}
Given any PPT adversary $\adv$ that is allowed to corrupt $\tcor$ participants and wins the unforgeability game against FROST,
and a simulating algorithm $\SIM$ which can simulate FROST,
then there exists a PPT AOMDL adversary $\ddv$ that can use $\adv$ as a subroutine to solve for its AOMDL challenge.

Concretely,
    \begin{align} \label{eq:ddv-dl}
    \begin{split}
        \advant{aomdl}{\ddv}(\lambda) & \geq
        \frac{\advant{uf}{\adv,\frost}(\lambda, n, t, t-1)^2}{\rototal}
        - \frac{\honetotal}{q}
    \end{split}
    \end{align}
where $\rototal =  \qnon + \qsigh + q_s + 1$.
\end{lemma}

\begin{proof}[Proof of Lemma~\ref{lem:frost-simulation}]
  Let $\A$ be an adversary playing the static unforgeability game as defined in Figure~\ref{fig:euf-cma-threshold},
  instantiated with FROST.
  $\A$ is allowed to query $\HashNon$ $\qnon$ number of times,
  $\HashSig$ $\qsigh$ number of times,
  and is allowed to make $q_s$ signing queries.
  Without loss of generality, we assume $\A$ queries $\HashSig$ on its forgery $(\threshpk, m^*, R^*)$.
  Finally, let $\rototal = \qnon + \qsigh + q_s + 1$.



  \noindent\textit{Setup.}
  $\SIM$ is initialized with parameters $\pp = (\Gr, g, p)$ and a set  of $2q_s+1$ AOMDL challenges $\{ X_0, \ldots, X_{2q_s}\}$,
  and can query the AOMDL oracle $\odlsol$ up to $q_s$ times.
  $\SIM$ then initializes the following values:
  \begin{enumerate}
    \item tables $\QNon \gets \emptyset$, $\QSig \gets \emptyset$ to simulate random oracles $\HashNon, \HashSig$;
    \item tables $Q_1 \gets \emptyset$ to maintain state during its simulation;
    \item tables $W, W'$ to maintain open and closed sessions,
    \item the table $Q_m \gets \emptyset$ to maintain the set of messages
queried by $\A$ to honest signers,
    \item the counter $\counter$ to keep track of which AOMDL challenges have yet been used.
  \end{enumerate}

  $\SIM$ runs $\A(\pp)$ on the public parameters $\pp$.
  $\A$ chooses the initial set of corrupted parties $\corrupt \gets \set{n}, \lvert \corrupt \rvert \leq \tcor$.
  $\SIM$ sets $\honest \gets [n]~\backslash~\corrupt$ and must reveal the secret keys of the corrupted parties to $\A$,
  which it does in the next step.

  \medskip

  \noindent\textit{Simulating KeyGen.}
    To simulate key generation where $\SIM$ is the trusted dealer,
    $\SIM$ does the following:
    \begin{enumerate}
    \item It sets $\pkgroup = X_0$.
    \item To simulate key generation for the signing keys,
    $\SIM$ first picks $t-1$ secret key shares for the corrupt parties
        $(\sk_i)_{i \in \corrupt} \rgets \Zq$,
    and defines their corresponding public keys
        $\pk_i \gets g^{\sk_i}$ for all $i \in \corrupt$.
    \item $\SIM$ then derives the public key shares
      $\pk_j, j \in \honest$ for the remaining honest parties in the exponent, as follows:
  \begin{align} \label{eq:keygen-sim}
    \pk_j = (\pk)^{L_0(j)} \cdot \prod_{i \in \corrupt} g^{\sk_i L_i(j)}
	\end{align}
        where $L_i(x)$ is the $i\supth$ Lagrange polynomial defined by $\corrupt \cup \{0\}$.
    \end{enumerate}


  Then,
  $\SIM$ runs
  $\adv^{ \BigO^{\Sign_1, \Sign_2} }(\state_A, \pkgroup, \{ \pkparty_j \}_{j \in \set{n}}, \{ \skparty_i \}_{i \in \corrupt})$,
  and responds to its oracle queries,
  as we describe next.

  \medskip
  \paragraph{Simulating Random Oracles.}
  To simulate $\adv$'s random oracle queries,
  $\SIM$ simply performs lazy sampling.
  It programs $\HashNon, \HashSig$ with values from its input
  $\{h_1, \ldots, h_{\rototal} \}$,
  as follows.

  \begin{itemize}
  \item{}\underline{$\HashNon:$} When $\A$ queries $\HashNon$ on $(k, \threshpk, m, \{   (i, R_i, S_i)\}_{i \in \coalition})$,
  To respond,
  $\SIM$ does the following:
  \begin{enumerate}
    \item First, $\SIM$ checks if $\QNon[(k, \threshpk, \allowbreak m,\allowbreak \{ (i, R_i, S_i)\}_{i \in \coalition})] \neq  \bot$.
      If not, $\SIM$ simply returns $\QNon[(k, \threshpk, \allowbreak m,\allowbreak \{ (i, R_i, S_i)\}_{i \in \coalition})] $,
    \item Otherwise, for all $j \in \coalition$, $\SIM$ does the following:
  \begin{enumerate}
    \item $\SIM$ sets $a_j \gets h_j $ and
      $\QNon[(j, \threshpk, m, \{(i, R_i, S_i)\}_{i \in \coalition})] = a_j$
      where $h_j$ is the next-unused hash value provided to $\SIM$ on initialization.
  \end{enumerate}
    \item Then, $\SIM$ derives $\aggR \gets \prod_{i \in \coalition} R_i \cdot S_i^{a_i}$.
    \item $\SIM$ checks if $\QSig[(\threshpk, m, \aggR)] = \bot$;
      if the check does not hold,
      $\SIM$ returns $\BadEventOne$.
    \item Otherwise, $\SIM$ sets $c \gets h_{j+1}$,
      where $h_j$ is the next-unused random oracle output,
      and stores $\QSig[(\pk, m, \aggR)] = c$.
    \item Finally, $\SIM$ returns $a_k$.
  \end{enumerate}


  \item{}\underline{$\HashSig:$} When $\A$ queries $\HashSig$ on $(\threshpk, m, \aggR)$,
  $\SIM$ checks whether $\QSig[(\threshpk, m, \aggR)] \neq \bot$;
  if the check does not hold,
  then
  $\SIM$ returns $\QSig[(\threshpk, m, \aggR)]$.
	Else, $\SIM$ sets $c \gets h_j$,
  where $h_j$ is the next-unused random oracle output,
  stores $\QSig[(\pk, m, \aggR)] = c$ and returns $c$.
  \end{itemize}


  \medskip


	\smallskip

  \paragraph{Simulating Signing Oracles.}
  To simulate signing oracles,
  $\SIM$ does as follows:

  \begin{itemize}
    \item{Round 1 ($\BigO^{\Sign}(k, \ssid)$):}
	When $\A$ queries $\BigO^{\Sign}$ on honest participant identifier $k \in \honest$ and session identifier $\ssid$,
  $\SIM$ does the following:
  \begin{enumerate}
  \item Checks if this signing session has already started by checking if $(k, \ssid) \in W$,
  returning $\bot$ if the check holds.
  \item Otherwise, it opens a new signing session for participant $k$ by storing $W \gets W \cup \{(k, \ssid) \} $.
  \item Next, $\SIM$ sets $R_k \gets X_\counter$ and $S_k \gets X_{\counter+1}$,
  where $\counter$ represents the next-unused value from the set of challenges $(X_1, \ldots, X_{2 q_S})$.
  It then increments $\counter \gets \counter + 2$.
  \item $\SIM$ then sets $\state_{k,\ssid} \gets (k, \ssid, R_k, S_k)$,
  and stores $Q_1[(k, \ssid)] = (R_k, S_k)$,
  \item Finally, $\B$ returns $(R_k, S_k)$ to $\A$.
  \end{enumerate}

\item{Round 2 ($\BigO^{\Sign'}(k, \ssid, m, \S, \{ (R_i, S_i) \}_{i \in \coalition})$):}
	When $\A$ queries $\BigO^{\Sign}$ for honest participant identifier $k \in \honest$,
  session identifier $\ssid$, message $m$, signing set $\coalition$,
  and set of nonce commitments $\{ (R_i, S_i) \}_{i \in \coalition}$,
  $\SIM$ does as follows:
  \begin{enumerate}
  \item Checks if this signing session has already closed by checking if $(k, \ssid) \in W'$,
  \item Checks if this signing session has not yet been opened, by if $(k, \ssid) \not\in W$,
  returning $\bot$ if the check holds.
\item Checks that $(k, \ssid, R_k, S_k) \in Q_1$ and $Q_2[(R_k, S_k)] = \bot$
    to ensure that they were indeed issued in Round 1 and have not yet been used,
    returning $\bot$ if the check does not hold.
  \item Otherwise, $\SIM$ follows the protocol honestly to derive each $a_i, i \in \coalition$,
  and then the group commitment $\aggR \gets \prod_{i \in \coalition} R_i \cdot {S_i}^{a_i}$
  and the challenge $c \gets \HashSig(\threshpk, m, \aggR)$.
  \item $\B$ then sets $Z_k \gets R_k \cdot S_k^{a_k} \cdot {\pk_k}^{c \lambda_k}$,
  and queries the AOMDL oracle $\odlsol$ on $Z_k$ and
  coefficients $(\alpha, ( \beta_i)_{i \in \set{2q_s + 1}})$,
  where each coefficient is set to zero,
  with the exception of
  $(\beta_0 = c \cdot \gamma, \beta_\counter = 1, \beta_{\counter+1} = a_i)$.
  Here, $(\counter, \counter+1)$ corresponds to the AOMDL challenges $(R_i, S_i)$ for this signing query,
  and $\gamma = \sum_{i \in \corrupt} \sk_i L_i'$,
  where $L_i'$ is the Lagrange coefficient for $i \in \corrupt$ associated with the set $\{ 0\} \cup \corrupt$,
  such that $\pk_k^c = \pk^{L_0' c} \cdot g^{c \gamma}$.
  \item $\SIM$ then receives $z_k$ in return from its AOMDL oracle,
    which is the discrete logarithm of $Z_k$.
  \end{enumerate}
  \item $\SIM$ then stores $Q_2[(R_k, S_k)] = (z_k, c \lambda_k)$.
  \item $\SIM$ then returns $z_k$ to $\A$.
  \end{itemize}

  \medskip

  \paragraph{Analysis of Simulation.}
	$\SIM$'s simulation of the random oracles, key generation, and signing is perfect,
  with exception of bad events.
  $\SIM$ responds to random oracle queries by lazy sampling;
  $\SIM$ aborts due to $\BadEvent$ with probability $1/q$ for each query to $\HashNon$,
  for a total of $\rototal$ possible queries.

  For key generation,
  $\A$ receives a valid secret key that is the point $i$ on the polynomial $f$,
  where $\pk_i = g^{f(i)}$.
  For signing, simulation is perfect because $\SIM$ receives $z_k$ from its AOMDL oracle,
  which is the discrete logarithm of $Z_k$,
  with respect to $R_k, S_k, \pk_k$.


\paragraph{Output.} At the end of the game,
$\adv$ outputs a forgery $(m^*, \sigma^* = (R^*, z^*))$.
If $g^{z^*} \neq R^* \cdot (\pkgroup)^{c^*}$ (meaning that the forgery is invalid),
$\SIM$ outputs $\bot$.
Without loss of generality,
we assume that $\adv$ queried $\HashSig$ on $(R^*, \pkgroup, m^*)$.


Otherwise,
if $\adv$'s forgery is valid,
$\SIM$ outputs the tuple $(j, \aux)$,
where $j$ is the index corresponding to $c^* = h_j$,
and $\aux=(m^*, \sigma^*)$ is the adversary's forgery with respect to $\pkgroup=\dlchal$.

\end{proof}



\begin{proof}[Proof of Lemma~\ref{lem:frost-reduction}]
  We now show that if there exists a PPT adversary $\adv$ against the
  unforgeability of FROST and an algorithm $\SIM$
  as described in Lemma~\ref{lem:frost-simulation} that simulates FROST to $\adv$,
  then we can define a PPT adversary $\ddv$ that can efficiently solve for an AOMDL challenge instance.

  $\ddv$ begins by accepting parameters $\pp = (\Gr, g, p)$ and a set  of $2q_s+1$ AOMDL challenges $\{ X_0, \ldots, X_{2q_s}\}$.
  It then chooses a $(t, n)$ such that $n \geq t$.

  $\ddv$ then executes the modified forking algorithm $\modgenfork^{\SIM}(X)$ as described in Section~\ref{mod-fork} and shown in Figure~\ref{fg:gen-fork-mod},
  providing as input the instance $X = (\pp, \{ X_0, \ldots, X_{2q_s}\}, n, t)$.

  $\modgenfork^{\SIM}(X)$ then either outputs $\bot$ when it fails or the value $(j, \aux)$ when it accepts.
  Corollary~\ref{cor:modified-fork} upper-bounds the probability that $\modgenfork^{\SIM}(X)$ will output $\bot$
  and Lemma~\ref{lem:simulation} upper-bounds the additional advantage of $\adv$ against $\SIM$.

   We next show that when $\modgenfork^{\SIM}(X)$ outputs an accepting output
   (i.e., it does not output $\bot$),
   then $\ddv$ can solve for its AOMDL challenge with perfect success.

 \paragraph{Extracting the Discrete Logarithm of $X_0$.}
   In the case that $\modgenfork^{\SIM}(X)$ outputs an accepting output,
   it outputs the tuple $(h_j, h_j', \auxstring, \auxstring')$.
   From these values,
   $\ddv$ then parses the following:

   \begin{align*}
     \begin{split}
       (m^*, \sigma^*)  \gets \auxstring,
       & \quad (m^{**}, \sigma^{**})  \gets \auxstring', \\
       (R^*, z^*)  \gets \sigma^*,
       & \quad (R^{**}, z^{**})  \gets \sigma^{**}, \\
       h_j = c^*, & \quad h_j'=c^{**}
     \end{split}
   \end{align*}

  Because $h_j, h_j'$ are with respect to the same index $j$ corresponding to
  $\adv$'s random query for $c^*$, $c^{**}$ in its first and second execution,
  then we know that $R^{*}=R^{**}$ and $m^{*}=m^{**}$.
   $\ddv$ can then solve for the discrete logarithm $y$ to its challenge $\dlchal = g^y$,
   as follows:

   \begin{equation}
     x_0 = \skgroup =  \frac{z^* - z^{**}}{c^* - c^{**}}
   \end{equation}

   Because $\modgenfork$ outputs $\bot$ in the case when any random oracle outputs collide (i.e., in the case when any $h_i = h_i'$),
   then we know that this equation is solveable.
